function getRainflow(data, point, from, to, axes)
{
    if(point == 0) return [];
    var idx1 = from;

    for(var i = from + 1; i < to - 1; i++)
    {
        if( axes > 0 )
        {
            if( data[i] > data[idx1] ) idx1 = i;
        }
        if( axes < 0 )
        {
            if( data[i] < data[idx1] ) idx1 = i;
        }
    }
    var idx2 = idx1;

    for(var i = idx1 + 1; i < to - 1; i++)
    {
        if( axes > 0 )
        {
            if( data[i] < data[idx2] ) idx2 = i;
        }
        if( axes < 0 )
        {
            if( data[i] > data[idx2] ) idx2 = i;
        }
    }

    var indexes = [];

    if( idx1 != from && idx2 != idx1 )
    {
        indexes.push(idx1, idx2);

        if( point > 1 )
        {
            indexes = indexes.concat(getRainflow(data, point - 1, from, idx1 + 1, axes));
            indexes = indexes.concat(getRainflow(data, point - 1, idx1, idx2 + 1, axes * -1));
            indexes = indexes.concat(getRainflow(data, point - 1, idx2, to, axes));
        }
    }

    return indexes;
}
